🎯 Άσκηση 1: Number Guessing Game

Στόχος: Να δημιουργήσεις ένα παιχνίδι μαντεψίματος αριθμών.

import random

def number_guessing_game():
    secret_number = random.randint(1, 100)
    attempts = 0
    
    print("🎮 Μάντεψε τον αριθμό (1-100)!")
    
    while True:
        try:
            guess = int(input("Δώσε τον αριθμό σου: "))
            attempts += 1
            
            if guess < secret_number:
                print("📈 Ψηλότερα!")
            elif guess > secret_number:
                print("📉 Χαμηλότερα!")
            else:
                print(f"🎉 Σωστά! Το βρήκες σε {attempts} προσπάθειες!")
                break
                
        except ValueError:
            print("⚠️ Παρακαλώ δώσε έναν έγκυρο αριθμό!")

# Καλείς τη συνάρτηση
number_guessing_game()
